@startuml Sequence_Predict_Performance
!theme plain
title Sequence Diagram - Predict Student Performance

actor Student
participant "Web Browser" as Browser
participant "Router" as Router
participant "PredictionController" as Controller
participant "PredictionService" as Service
participant "KNNPredictor" as KNN
participant "StudentModel" as SModel
participant "GradeModel" as GModel
participant "Database" as DB

Student -> Browser : Request Prediction
Browser -> Router : HTTP Request
Router -> Controller : dispatch()
Controller -> Controller : requireLogin()
Controller -> Service : predictPerformance(studentId)
Service -> SModel : findById(studentId)
SModel -> DB : SELECT student data
DB --> SModel : Student data
SModel --> Service : Student object

Service -> GModel : getGradesByStudent(studentId)
GModel -> DB : SELECT grades
DB --> GModel : Grades data
GModel --> Service : Grades array

Service -> Service : prepareTrainingData()
Service -> DB : SELECT all students with grades
DB --> Service : Training data

Service -> KNN : loadTrainingData(trainingData)
Service -> Service : getStudentFeatures(studentId)
Service -> KNN : predict(features)

KNN -> KNN : normalizeFeatures()
KNN -> KNN : calculateDistances()
KNN -> KNN : findKNearestNeighbors()
KNN -> KNN : majorityVote()
KNN --> Service : Prediction result

Service -> Service : identifyRiskFactors()
Service -> Service : savePrediction()
Service -> SModel : updateRiskLevel()
SModel -> DB : UPDATE student
Service --> Controller : Prediction result

Controller -> Browser : JSON Response
Browser --> Student : Display Prediction

@enduml

@startuml Sequence_Login
!theme plain
title Sequence Diagram - User Login

actor User
participant "Web Browser" as Browser
participant "Router" as Router
participant "AuthController" as Controller
participant "UserModel" as Model
participant "Database" as DB
participant "Session" as Session

User -> Browser : Enter credentials
Browser -> Router : POST /login
Router -> Controller : dispatch()
Controller -> Controller : login()

alt Valid credentials
    Controller -> Model : findByEmail(email)
    Model -> DB : SELECT user
    DB --> Model : User data
    Model --> Controller : User object
    
    Controller -> Model : verifyPassword(password, hash)
    Model --> Controller : true
    
    Controller -> Session : Set session variables
    Controller -> Browser : Redirect to Dashboard
    Browser --> User : Dashboard page
else Invalid credentials
    Controller -> Browser : Error message
    Browser --> User : Login page with error
end

@enduml

@startuml Sequence_Alert_Creation
!theme plain
title Sequence Diagram - Alert Creation

participant "PredictionService" as Service
participant "StudentModel" as SModel
participant "AlertModel" as AModel
participant "Database" as DB

Service -> Service : predictPerformance()
Service -> Service : identifyRiskFactors()
Service -> SModel : findById(studentId)
SModel -> DB : SELECT student
DB --> SModel : Student data
SModel --> Service : Student object

alt Risk level is high
    Service -> Service : createRiskAlert()
    Service -> AModel : getAlertsByStudent()
    AModel -> DB : SELECT alerts
    DB --> AModel : Existing alerts
    AModel --> Service : Alerts array
    
    alt No active alert exists
        Service -> AModel : createAlert()
        AModel -> DB : INSERT alert
        DB --> AModel : Alert created
        AModel --> Service : Success
    else Active alert exists
        Service -> Service : Skip creation
    end
end

@startuml State_Diagram
!theme plain
title State Diagram - Student Risk Level Management

[*] --> Initial : System Start

state Initial {
    [*] --> NoData
    NoData : No student data available
}

Initial --> DataCollection : Student enrolled

state DataCollection {
    [*] --> CollectingGrades
    CollectingGrades : Collecting grades and attendance
    CollectingGrades --> SufficientData : Enough data collected
    SufficientData : GPA, attendance, grades available
}

DataCollection --> Prediction : Trigger prediction

state Prediction {
    [*] --> LoadingTrainingData
    LoadingTrainingData : Load training data from database
    LoadingTrainingData --> CalculatingFeatures
    CalculatingFeatures : Calculate student features
    CalculatingFeatures --> RunningKNN
    RunningKNN : Run KNN algorithm
    RunningKNN --> DeterminingRisk
    DeterminingRisk : Determine risk level
}

Prediction --> LowRisk : Risk = Low
Prediction --> MediumRisk : Risk = Medium
Prediction --> HighRisk : Risk = High

state LowRisk {
    [*] --> Monitoring
    Monitoring : Monitor student performance
    Monitoring --> LowRisk : Performance maintained
    Monitoring --> MediumRisk : Performance declined
}

state MediumRisk {
    [*] --> Monitoring
    Monitoring : Monitor student performance
    Monitoring --> LowRisk : Performance improved
    Monitoring --> HighRisk : Performance declined
    Monitoring --> MediumRisk : Performance stable
}

state HighRisk {
    [*] --> AlertCreated
    AlertCreated : Alert created for instructor
    AlertCreated --> Intervention
    Intervention : Intervention actions taken
    Intervention --> Monitoring
    Monitoring : Monitor after intervention
    Monitoring --> MediumRisk : Performance improved
    Monitoring --> HighRisk : Performance still poor
    Monitoring --> Escalation : No improvement
}

state Escalation {
    [*] --> AdminNotified
    AdminNotified : Admin notified
    AdminNotified --> AdditionalActions
    AdditionalActions : Additional actions taken
}

Escalation --> HighRisk : Continue monitoring
HighRisk --> MediumRisk : Performance improved
MediumRisk --> LowRisk : Performance improved

LowRisk --> [*] : Student graduated
MediumRisk --> [*] : Student graduated
HighRisk --> [*] : Student graduated

note right of HighRisk
    When risk level changes to High:
    - Alert is automatically created
    - Instructor is notified
    - Prediction is saved
end note

note right of Prediction
    KNN Algorithm:
    1. Load training data
    2. Normalize features
    3. Calculate distances
    4. Find k nearest neighbors
    5. Majority vote for risk level
end note

@enduml

@startuml State_Diagram_User_Session
!theme plain
title State Diagram - User Session Management

[*] --> LoggedOut : System Start

state LoggedOut {
    [*] --> ViewingPublicPages
    ViewingPublicPages : Can view public pages
    ViewingPublicPages --> AttemptingLogin : User clicks login
}

LoggedOut --> Authenticating : Submit credentials

state Authenticating {
    [*] --> ValidatingCredentials
    ValidatingCredentials : Check email and password
    ValidatingCredentials --> Valid : Credentials correct
    ValidatingCredentials --> Invalid : Credentials incorrect
}

Authenticating --> LoggedIn : Valid credentials
Authenticating --> LoggedOut : Invalid credentials

state LoggedIn {
    [*] --> SessionActive
    SessionActive : Session variables set
    
    state RoleBased {
        [*] --> AdminDashboard
        [*] --> InstructorDashboard
        [*] --> StudentDashboard
        
        AdminDashboard : Full system access
        InstructorDashboard : Course and student management
        StudentDashboard : View own data
    }
    
    SessionActive --> RoleBased : Redirect to dashboard
    RoleBased --> UsingSystem : User interacts
    UsingSystem : Using various features
    UsingSystem --> SessionActive : Return to dashboard
}

LoggedIn --> SessionExpired : Timeout or inactivity
LoggedIn --> LoggedOut : User logs out

state SessionExpired {
    [*] --> SessionCleared
    SessionCleared : Session variables cleared
}

SessionExpired --> LoggedOut : Redirect to login

LoggedOut --> [*] : System shutdown

note right of Authenticating
    Validation Process:
    1. Check if user exists
    2. Verify password hash
    3. Check user status (active/inactive)
    4. Set session variables
end note

note right of LoggedIn
    Session Variables:
    - user_id
    - role (admin/instructor/student)
    - name
    - email
end note

@enduml
